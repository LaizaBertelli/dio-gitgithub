# Comandos utilizados para a criação do repositório

    mkdir git-github && cd git-github
    git init
    touch README.md
    git add .
    git commit -m "Init repo"
    git branch -M main
    git remote add origin git@github.com:LaizaBertelli/dio-gitgithub.git
    git push -u origin main
